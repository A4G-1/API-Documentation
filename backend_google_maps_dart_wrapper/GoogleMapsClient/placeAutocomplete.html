Future<List<Suggestion>> placeAutocomplete({
  required String input,
}) async {
  http.BaseRequest req = http.Request(
    'get',
    Uri.parse(
      'https://maps.googleapis.com/maps/api/place/autocomplete/json?input=$input&key=$apiKey',
    ),
  );
  http.StreamedResponse res = await client.send(req);
  if (res.statusCode == 200) {
    Map<String, dynamic> data =
        jsonDecode(await res.stream.bytesToString()) as Map<String, dynamic>;
    if (data['status'] == 'OK') {
      return data['predictions'].map<Suggestion>((prediction) {
        return Suggestion(
          prediction['place_id'],
          prediction['description'],
        );
      }).toList();
    }
    if (data['status'] == 'ZERO_RESULTS') {
      return [];
    }
    print(data);
    throw Exception(data['error_message']);
  } else {
    throw Exception('Failed to load suggestions');
  }
}
