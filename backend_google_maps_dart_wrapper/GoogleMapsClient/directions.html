Future<Map<String, dynamic>?> directions({
  required String origin,
  required String destination,
  List<String>? waypoints,
}) async {
  try {
    http.BaseRequest req = http.Request(
      'get',
      Uri.parse(
        'https://maps.googleapis.com/maps/api/directions/json?key=$apiKey&origin=$origin&destination=$destination&units=imperial&waypoints=${waypoints == null ? '' : waypoints.join('|')}',
      ),
    );
    http.StreamedResponse res = await client.send(req);
    Map<String, dynamic> data =
        jsonDecode(await res.stream.bytesToString()) as Map<String, dynamic>;
    return data;
  } catch (e) {
    print(e);
  }
  return null;
}
